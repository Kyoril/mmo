        return;

    if (!target->IsUnit())
    {
        WLOG("TRIGGER_ACTION_SET_STAND_STATE: Needs a unit target - action ignored");
        return;
    }
    const uint32 standState = GetActionData(action, 0);
    // ... validate and apply ...
    target->AsUnit().SetStandState(static_cast<unit_stand_state::Type>(standState));
}
