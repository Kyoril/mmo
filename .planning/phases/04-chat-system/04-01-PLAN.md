<objective>
Fix the server-side chat handler so Whisper and Raid messages are fully routed.

Purpose: The packet parser for `OnChatMessage()` currently only reads `chatType` + `message` — whisper target name bytes remain unconsumed in the buffer, corrupting subsequent reads and silently dropping whispers. Raid has a stub that logs a warning and discards messages. This plan closes both gaps at the lowest level and adds the `ChatType::System` enum value used by the "player not found" error path.

Output:
- `chat_type.h` with `ChatType::System` appended after `UnitEmote`
- `player.cpp` with correct target-name parsing for Whisper/Channel packets, a working Whisper case (lookup → route, or error), and a working Raid case (group broadcast)
</objective>

```cpp
case ChatType::Whisper:
{
    if (targetName.empty())
    {
        WLOG("Whisper received with empty target name, ignoring.");
        break;
    }

    Player* target = m_manager.GetPlayerByCharacterName(targetName);
    if (target == nullptr)
    {
        // Target not found — send system error back to sender; skip DB log for failed whispers
        const std::string errorMsg = "No player named '" + targetName + "' is currently online.";
        SendPacket([this, &errorMsg](game::OutgoingPacket &outPacket)
        {
            outPacket.Start(game::realm_client_packet::ChatMessage);
            outPacket
                << io::write_packed_guid(0)
                << io::write<uint8>(ChatType::System)
                << io::write_range(errorMsg) << io::write<uint8>(0)
                << io::write<uint8>(0)
                << io::write<uint8>(0);
            outPacket.Finish();
        });
        return PacketParseResult::Pass;
    }

    target->SendPacket([this, &message, chatType](game::OutgoingPacket &outPacket)
    {
        outPacket.Start(game::realm_client_packet::ChatMessage);
        outPacket
            << io::write_packed_guid(m_characterData->characterId)
            << io::write<uint8>(chatType)
            << io::write_range(message) << io::write<uint8>(0)
            << io::write<uint8>(0)
            << io::write<uint8>(0);
        outPacket.Finish();
    });
}
break;
```
